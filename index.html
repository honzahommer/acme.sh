#!/usr/bin/env sh #https://git.io/acme.sh _exists() { cmd="$1" if [ -z "$cmd" ] ; then echo "Usage: _exists cmd" return 1 fi if type command >/dev/null 2>&1 ; then command -v "$cmd" >/dev/null 2>&1 else type "$cmd" >/dev/null 2>&1 fi return "$?" } _truthy() { val="$(echo "$1" | tr '[:upper:]' '[:lower:]')" [ -n "$val" ] && [ "$val" != 0 ] && [ "$val" != false ] } if _truthy "$HTTPS_INSECURE"; then _CURL_OPTS="--insecure" _WGET_OPTS="--no-check-certificate" else _CURL_OPTS= _WGET_OPTS= fi if _truthy "$ACME_USE_WGET"; then _USE_WGET=1 else _USE_WGET=0 fi _ACME="https://raw.githubusercontent.com/honzahommer/acme.sh/feat/installonline-arguments/acme.sh" _CURL="curl $_CURL_OPTS $_ACME" _WGET="wget $_WGET_OPTS -O - $_ACME" if _exists curl && [ -z "$_USE_WGET" ]; then _CMND="$_CURL" elif _exists wget ; then _CMND="$_WGET" else echo "Sorry, you must have curl or wget installed first." echo "Please install either of them and try again." exit 2 fi $_CMND | ACME_USE_WGET=$_USE_WGET INSTALLONLINE=1 sh -s -- "$@"
